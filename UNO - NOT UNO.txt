n = int(input())
s = str(n)
f= True
while(f==True):
    sum = 0
    for i in s:
        sum += int(i)
    if(len(str(s))==1):
        f = False
        if(int(s)==1):
            print("UNO")
        else:
            print("NOT UNO")
            
    s = str(sum)
