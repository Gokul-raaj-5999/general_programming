n = int(input())
start = []
end = []
for i in range(n):
    start.append(int(input()))
for i in range(n):
    end.append(int(input()))
l = []
for i in range(n):
    count = 0
    lend = end[i]
    for j in range(n):
        if(j==0):
            count+=1
        else:
            if(start[j]>lend):
                lend = end[j]
                count+=1
    l.append(count)
print(max(l))
